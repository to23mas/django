Example Requests

Get All Books

{
  "jsonrpc": "2.0",
  "method": "get_books",
  "params": {},
  "id": 1
}

Get Single Book

{
  "jsonrpc": "2.0",
  "method": "get_book",
  "params": {"id": 1},
  "id": 2
}

Create Book

{
  "jsonrpc": "2.0",
  "method": "create_book",
  "params": {
    "title": "New Book",
    "author": "Author Name",
    "isbn": "1234567890",
    "pages": 200,
    "published_date": "2024-03-16"
  },
  "id": 3
}

Update Book

{
  "jsonrpc": "2.0",
  "method": "update_book",
  "params": {
    "id": 1,
    "title": "Updated Title",
    "author": "Updated Author"
  },
  "id": 4
}

Delete Book

{
  "jsonrpc": "2.0",
  "method": "delete_book",
  "params": {"id": 1},
  "id": 5
}
