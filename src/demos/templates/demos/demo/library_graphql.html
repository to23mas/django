Library GraphQL API

Example Queries

Get All Books

query {
  books {
    id
    title
    author
  }
}

Get Single Book

query {
  book(id: 1) {
    title
    author
    isbn
  }
}

Create Book

mutation {
  createBook(
    title: "New Book"
    author: "Author Name"
    isbn: "1234567890"
    pages: 200
    publishedDate: "2024-03-16"
  ) {
    id
    title
  }
}
